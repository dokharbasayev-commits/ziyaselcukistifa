  <div id="wrapper">
    <!-- Videonun adını burada değiştirirsin -->
    <video id="scareVid"
           src="m2-res_720p.mp4"
           autoplay
           playsinline>
    </video>
    <div id="tap-hint">Tam ekran ve ses için ekrana dokun</div>
  </div>

  <script>
    const video = document.getElementById('scareVid');
    const hint  = document.getElementById('tap-hint');

    // Ses açık olsun
    video.volume = 1.0;

    // Sayfa açılınca yine de play etmeye zorla
    window.addEventListener('load', () => {
      video.play().catch(() => {
        // Bazı tarayıcılar user interaction isteyebilir, sorun değil
      });
    });

    // Kullanıcı ilk kez ekrana tıklayınca:
    const goFull = () => {
      // Tam ekran dene
      if (video.requestFullscreen) {
        video.requestFullscreen().catch(() => {});
      } else if (video.webkitRequestFullscreen) {
        video.webkitRequestFullscreen();
      }

      // Videoyu emin olmak için tekrar oynat
      video.play().catch(() => {});

      // Yazıyı kaldır
      hint.style.display = "none";

      // Bir daha dinleme
      window.removeEventListener('click', goFull);
      window.removeEventListener('touchstart', goFull);
    };

    window.addEventListener('click', goFull);
    window.addEventListener('touchstart', goFull);
  </script>
